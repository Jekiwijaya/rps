import * as React from 'react';
 
import { Provider, initialState } from './ComponentStateContext';
import { ComponentStateState } from '../types/ComponentState';
 
export default class ComponentStateProvider extends React.Component<any, ComponentStateState> {
  private mount: boolean = false;
  public state: ComponentStateState = initialState;
 
  constructor(props: any) {
    super(props);
    this.mount = true;
  }
 
  public componentWillUnmount() {
    this.mount = false;
  }
 
  public render() {
    return <Provider value={this.getContext()}>{this.props.children}</Provider>;
  }
 
  private setLoading = (isLoading: boolean) => {
    if (!this.mount) return;
    const loadingValue: number = isLoading ? 1 : -1;
    this.setState(({ loadingCount }) => ({
      isLoading: loadingCount + loadingValue > 0,
      loadingCount: loadingCount + loadingValue,
    }));
  };
 
  private setError = (error: any) => {
    if (!this.mount) return;
    this.setResult(null);
    this.setState({
      result: null,
      isError: !!error,
      error,
    });
  };
 
  private setResult = (result: any) => {
    if (!this.mount) return;
    this.setError(null);
    this.setState({
      result,
    });
  };
 
  private getContext() {
    return {
      ...this.state,
      setLoading: this.setLoading,
      setError: this.setError,
      setResult: this.setResult,
    };
  }
}
