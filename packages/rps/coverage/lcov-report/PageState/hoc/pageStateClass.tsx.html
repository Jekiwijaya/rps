import * as React from 'react';
 
import PageStateConsumer from '../PageStateConsumer';
import * as hoistNonReactStatics from 'hoist-non-react-statics';
import { PageStateContext } from '../../types/PageState';
 
export default function pageStateClass() {
  return (WrapperComponent: React.ComponentType<any>) => {
    let Comp: React.ComponentType<any> = class extends React.Component<any> {
      public render() {
        return (
          <PageStateConsumer>
            {(propsConsumer: PageStateContext) => <WrapperComponent pageState={propsConsumer} {...this.props} />}
          </PageStateConsumer>
        );
      }
    };
    Comp = hoistNonReactStatics(Comp, WrapperComponent);
    return Comp;
  };
}
